# Actor-Critic Network
class ActorCritic(nn.Module):
    def __init__(self, input_dim, action_dim, continuous=False):
        super(ActorCritic, self).__init__()
        self.continuous = continuous
        
        # Shared base network
        self.base = nn.Sequential(
            nn.Linear(input_dim, 256),
            nn.ReLU(),
            nn.Linear(256, 256),
            nn.ReLU()
        )
        
        # Actor head
        if continuous:
            self.mu = nn.Linear(256, action_dim)
            self.log_std = nn.Parameter(torch.zeros(action_dim))
        else:
            self.actor = nn.Linear(256, action_dim)
        
        # Critic head
        self.critic = nn.Linear(256, 1)
        
    def forward(self, x):
        base_out = self.base(x)
        if self.continuous:
            mu = torch.tanh(self.mu(base_out))
            std = torch.exp(self.log_std)
            dist = distributions.Normal(mu, std)
        else:
            logits = self.actor(base_out)
            dist = distributions.Categorical(logits=logits)
            
        value = self.critic(base_out)
        return dist, value

# PPO Agent
class PPOAgent:
    def __init__(self, env, lr=3e-4, gamma=0.99, clip_eps=0.2, update_epochs=10, batch_size=64):
        self.env = env
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        self.gamma = gamma
        self.clip_eps = clip_eps
        self.update_epochs = update_epochs
        self.batch_size = batch_size
        
        # Initialize network
        self.continuous = isinstance(env.action_space, gym.spaces.Box)
        input_dim = env.observation_space.shape[0]
        action_dim = env.action_space.shape[0] if self.continuous else env.action_space.n
        self.model = ActorCritic(input_dim, action_dim, self.continuous).to(self.device)
        self.optimizer = optim.Adam(self.model.parameters(), lr=lr)
        
        # Buffers
        self.states = []
        self.actions = []
        self.log_probs = []
        self.rewards = []
        self.values = []
        self.dones = []
        
    def act(self, state):
        state = torch.FloatTensor(state).to(self.device)
        dist, value = self.model(state)
        if self.continuous:
            action = dist.sample()
            log_prob = dist.log_prob(action).sum(-1).detach()
        else:
            action = dist.sample()
            log_prob = dist.log_prob(action).detach()
        return action.cpu().numpy(), log_prob, value
        
    def store(self, state, action, log_prob, reward, value, done):
        self.states.append(state)
        self.actions.append(action)
        self.log_probs.append(log_prob)
        self.rewards.append(reward)
        # Detach the value here to avoid reusing the computation graph
        self.values.append(value.detach())
        self.dones.append(done)
        
    def compute_returns(self):
        returns = []
        R = 0
        for r, d in zip(reversed(self.rewards), reversed(self.dones)):
            if d:
                R = 0
            R = r + self.gamma * R
            returns.insert(0, R)
        returns = torch.FloatTensor(returns).to(self.device)
        return (returns - returns.mean()) / (returns.std() + 1e-8)
        
    def update(self):
        states = torch.FloatTensor(np.array(self.states)).to(self.device)
        actions = torch.FloatTensor(np.array(self.actions)).to(self.device)
        old_log_probs = torch.stack(self.log_probs).to(self.device)
        returns = self.compute_returns()
        values = torch.stack(self.values).squeeze().to(self.device)
        advantages = returns - values
        advantages = (advantages - advantages.mean()) / (advantages.std() + 1e-8)
        
        for _ in range(self.update_epochs):
            for idx in range(0, len(states), self.batch_size):
                batch_idx = slice(idx, min(idx+self.batch_size, len(states)))
                batch_states = states[batch_idx]
                batch_actions = actions[batch_idx]
                batch_old_log_probs = old_log_probs[batch_idx]
                batch_advantages = advantages[batch_idx]
                batch_returns = returns[batch_idx]
                
                dist, value = self.model(batch_states)
                if self.continuous:
                    new_log_probs = dist.log_prob(batch_actions).sum(-1)
                else:
                    new_log_probs = dist.log_prob(batch_actions)
                    
                ratio = torch.exp(new_log_probs - batch_old_log_probs)
                surr1 = ratio * batch_advantages
                surr2 = torch.clamp(ratio, 1-self.clip_eps, 1+self.clip_eps) * batch_advantages
                actor_loss = -torch.min(surr1, surr2).mean()
                value_loss = nn.MSELoss()(value.squeeze(), batch_returns)
                loss = actor_loss + 0.5 * value_loss
                
                self.optimizer.zero_grad()
                loss.backward()
                nn.utils.clip_grad_norm_(self.model.parameters(), 0.5)
                self.optimizer.step()
                
        # Clear buffers after update
        self.states, self.actions, self.log_probs, self.rewards, self.values, self.dones = [], [], [], [], [], []
        
    def evaluate(self, num_episodes=10):
        eval_rewards = []
        for episode in range(num_episodes):
            state, _ = self.env.reset() if isinstance(self.env.reset(), tuple) else self.env.reset()
            state = np.array(state)
            episode_reward = 0
            done = False
            
            while not done:
                action, _, _ = self.act(state)
                step_result = self.env.step(action)
                state, reward, terminated, truncated, _ = step_result
                state = np.array(state)
                done = terminated or truncated
                episode_reward += reward
                
            eval_rewards.append(episode_reward)
            print(f"Evaluation Episode {episode+1}, Reward: {episode_reward:.2f}")
            
        return eval_rewards

def plot_rewards(env_name, train_rewards, eval_rewards):
    plt.figure(figsize=(12,6))
    plt.subplot(1,2,1)
    plt.plot(train_rewards, label='Training Reward')
    plt.title(f'{env_name} Training Rewards')
    plt.xlabel('Episode')
    plt.ylabel('Total Reward')
    plt.grid(True)
    plt.legend()
    
    plt.subplot(1,2,2)
    plt.plot(eval_rewards, label='Evaluation Reward', marker='o')
    plt.title(f'{env_name} Evaluation Rewards (10 Episodes)')
    plt.xlabel('Episode')
    plt.ylabel('Total Reward')
    plt.grid(True)
    plt.legend()
    
    plt.tight_layout()
    plt.savefig(f'{env_name}_rewards.png')
    plt.close()

def train_ppo(env_name, max_episodes=500, target_reward=None):
    env = gym.make(env_name)
    agent = PPOAgent(env)
    
    recent_rewards = deque(maxlen=100)
    episode_rewards = []
    
    for episode in range(max_episodes):
        state, _ = env.reset() if isinstance(env.reset(), tuple) else env.reset()
        state = np.array(state)
        episode_reward = 0
        
        while True:
            action, log_prob, value = agent.act(state)
            step_result = env.step(action)
            next_state, reward, terminated, truncated, _ = step_result
            next_state = np.array(next_state)
            done = terminated or truncated
            
            agent.store(state, action, log_prob, reward, value, done)
            state = next_state
            episode_reward += reward
            
            if done:
                break
                
        agent.update()
        recent_rewards.append(episode_reward)
        episode_rewards.append(episode_reward)
        
        print(f"Episode {episode+1}, Reward: {episode_reward:.2f}, Avg Reward: {np.mean(recent_rewards):.2f}")
        
        if target_reward and np.mean(recent_rewards) > target_reward:
            print(f"Target reward {target_reward} achieved!")
            break
            
    print(f"\nEvaluating {env_name} for 10 episodes")
    eval_rewards = agent.evaluate(num_episodes=10)
    
    # Save training plot rewards
    plot_rewards(env_name, episode_rewards, eval_rewards)
    # Save model weights to a .pth file
    torch.save(agent.model.state_dict(), f"{env_name}_ppo.pth")
    print(f"Model saved as {env_name}_ppo.pth")
    
    env.close()
    return episode_rewards, eval_rewards

import torch.nn as nn

# Neural network for policy and value function
class PolicyValueNetwork(nn.Module):
    def __init__(self, state_size, action_size, hidden_size=64):
        super(PolicyValueNetwork, self).__init__()
        self.layer1 = nn.Linear(state_size, hidden_size)
        self.layer2 = nn.Linear(hidden_size, hidden_size)
        self.action_mean = nn.Linear(hidden_size, action_size)
        self.log_variance = nn.Parameter(torch.zeros(action_size))
        self.value_output = nn.Linear(hidden_size, 1)

    def forward(self, state):
        x = F.relu(self.layer1(state))
        x = F.relu(self.layer2(x))
        mean = self.action_mean(x)
        std = self.log_variance.exp()
        value = self.value_output(x).squeeze(-1)
        return mean, std, value

    def choose_action(self, state, deterministic=False):
        mean, std, value = self.forward(state)
        distribution = torch.distributions.Normal(mean, std)
        action = mean if deterministic else distribution.sample()
        log_probability = distribution.log_prob(action).sum(dim=-1)
        return action, log_probability

    def assess_actions(self, states, actions):
        mean, std, value = self.forward(states)
        distribution = torch.distributions.Normal(mean, std)
        log_probabilities = distribution.log_prob(actions).sum(dim=-1)
        entropy = distribution.entropy().sum(dim=-1)
        return actions, log_probabilities, entropy

    def estimate_value(self, state):
        _, _, value = self.forward(state)
        return value

# Proximal Policy Optimization (PPO) implementation
class PPOTrainer:
    def __init__(self, policy_network, environment, lr=3e-4, rollout_steps=2048, minibatches=32, epochs=10,
                 discount=0.99, gae_factor=0.99, clip_range=0.2, entropy_weight=0.01, value_weight=0.5):
        self.policy = policy_network
        self.env = environment
        self.optimizer = optim.Adam(self.policy.parameters(), lr=lr)
        self.rollout_steps = rollout_steps
        self.minibatches = minibatches
        self.epochs = epochs
        self.discount = discount
        self.gae_factor = gae_factor
        self.clip_range = clip_range
        self.entropy_weight = entropy_weight
        self.value_weight = value_weight

    def calculate_gae(self, rewards, values, terminals):
        advantage = 0
        returns = []
        values.append(0.0)  # Append final value for terminal state
        for t in reversed(range(len(rewards))):
            delta = rewards[t] + self.discount * values[t + 1] * (1 - terminals[t]) - values[t]
            advantage = delta + self.discount * self.gae_factor * (1 - terminals[t]) * advantage
            returns.insert(0, advantage + values[t])
        return returns

    def train_step(self):
        state, _ = self.env.reset()
        states, actions, log_probs, rewards, terminals, values = [], [], [], [], [], []

        episode_rewards = []
        current_reward = 0

        # Collect experience
        for _ in range(self.rollout_steps):
            state_tensor = torch.tensor(state, dtype=torch.float32)
            action, log_prob = self.policy.choose_action(state_tensor)
            value = self.policy.estimate_value(state_tensor)

            next_state, reward, done, truncated, _ = self.env.step(action.detach().numpy())
            terminal = done or truncated

            states.append(state)
            actions.append(action.detach().numpy())
            log_probs.append(log_prob.detach().numpy())
            rewards.append(reward)
            terminals.append(terminal)
            values.append(value.detach().numpy())
            current_reward += reward

            if terminal:
                episode_rewards.append(current_reward)
                current_reward = 0
                next_state, _ = self.env.reset()

            state = next_state

        # Compute returns using Generalized Advantage Estimation (GAE)
        returns = self.calculate_gae(rewards, values, terminals)

        # Convert lists to tensors
        states = torch.tensor(np.array(states), dtype=torch.float32)
        actions = torch.tensor(np.array(actions), dtype=torch.float32)
        old_log_probs = torch.tensor(np.array(log_probs), dtype=torch.float32)
        returns = torch.tensor(np.array(returns), dtype=torch.float32)
        old_values = torch.tensor(np.array(values), dtype=torch.float32)

        # Normalize returns
        returns = (returns - returns.mean()) / (returns.std() + 1e-8)

        batch_size = self.rollout_steps
        minibatch_size = batch_size // self.minibatches

        # Update policy and value function
        for _ in range(self.epochs):
            for start in range(0, batch_size, minibatch_size):
                end = start + minibatch_size
                mb_states = states[start:end]
                mb_actions = actions[start:end]
                mb_log_probs = old_log_probs[start:end]
                mb_returns = returns[start:end]
                mb_values = old_values[start:end]

                _, new_log_probs, entropy = self.policy.assess_actions(mb_states, mb_actions)
                new_values = self.policy.estimate_value(mb_states)

                ratios = torch.exp(new_log_probs - mb_log_probs)
                advantages = mb_returns - mb_values
                surrogate1 = ratios * advantages
                surrogate2 = torch.clamp(ratios, 1 - self.clip_range, 1 + self.clip_range) * advantages
                policy_loss = -torch.min(surrogate1, surrogate2).mean()
                value_loss = nn.MSELoss()(new_values, mb_returns)
                entropy_loss = -entropy.mean()

                total_loss = policy_loss + self.value_weight * value_loss + self.entropy_weight * entropy_loss

                self.optimizer.zero_grad()
                total_loss.backward()
                self.optimizer.step()

        return {
            'policy_loss': policy_loss.item(),
            'value_loss': value_loss.item(),
            'episode_rewards': episode_rewards
        }
